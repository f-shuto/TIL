```ruby
n = gets.to_i
as = gets.split.map(&:to_i)

ans = 0
r = 0
checked = Array.new(n+10, false)
(0..n-1).each do |l|
  # 左端の値を利用済リストに格納
  checked[as[l]] = true
  # 右端を伸ばせるだけ伸ばす
  while r < n-1 && !checked[as[r+1]]
    r += 1
    # 右に伸ばした先の値を利用済リストに格納
    checked[as[r]] = true
  end
  ans = [ans, r-l+1].max
  checked[as[l]] = false
  if l == r
    r += 1
  end
end
puts ans
```
